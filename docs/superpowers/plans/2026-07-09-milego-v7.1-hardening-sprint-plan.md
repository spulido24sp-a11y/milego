import { bus } from './index.js';

export async function replayEvent(eventLogId) {
  const original = await db('event_logs').where({ id: eventLogId }).first();
  if (!original) throw new Error('Event not found');

  // Safeguard 1: antigüedad máxima (7 días)
  const MAX_AGE_DAYS = parseInt(process.env.MAX_REPLAY_AGE_DAYS, 10) || 7;
  const ageDays = (Date.now() - new Date(original.created_at).getTime()) / 86400000;
  if (ageDays > MAX_AGE_DAYS) {
    throw new Error(`Event too old (${Math.round(ageDays)} days, max ${MAX_AGE_DAYS})`);
  }

  // Parse payload
  const parsed = typeof original.payload === 'string' ? JSON.parse(original.payload) : original.payload;

  // Safeguard 2: version check (si hay contrato, validar)
  // (por ahora solo advertir si version cambió)

  // Safeguard 3: re-emitir con flag de replay (NO ejecuta webhooks)
  const newLog = await bus.emit(original.event_name, parsed.data, {
    ...parsed.meta,
    replayedFrom: original.id,
    isReplay: true,        // handlers deben respetar esto (NO ejecutar webhooks)
    correlationId: parsed.meta?.correlationId,
  });

  return newLog;
}

   b. `dequeue()` — atómico con `FOR UPDATE SKIP LOCKED`:

```javascript
export async function dequeue() {
  const [job] = await db('jobs')
    .where('status', 'pending')
    .where('available_at', '<=', db.fn.now())
    .orderBy('created_at', 'asc')
    .limit(1)
    .forUpdate()
    .skipLocked()
    .update({
      status: 'processing',
      locked_by: `worker-${require('crypto').randomUUID()}`,
      locked_at: db.fn.now(),
      started_at: db.fn.now(),
    })
    .returning('*');

  return job || null;
}
```

   c. `fail()` — insertar en `failed_jobs` al alcanzar `max_attempts`:

```javascript
export async function fail(jobId, reason) {
  const job = await db('jobs').where({ id: jobId }).first();
  const attempts = (job.attempts || 0) + 1;
  const maxAttempts = job.max_attempts || 5;

  if (attempts >= maxAttempts) {
    // Dead letter: mover a failed_jobs
    await db('failed_jobs').insert({
      original_job_id: jobId,
      type: job.type,
      payload: typeof job.payload === 'string' ? job.payload : JSON.stringify(job.payload),
      failed_reason: reason,
      attempts,
      failed_at: db.fn.now(),
    });

    await db('jobs').where({ id: jobId }).update({
      status: 'failed',
      failed_reason: reason,
      attempts,
    });
  } else {
    const backoffDelay = getBackoffDelay(attempts);
    await db('jobs').where({ id: jobId }).update({
      status: 'pending',
      attempts,
      available_at: new Date(Date.now() + backoffDelay * 1000),
      failed_reason: reason,
      locked_by: null,
      locked_at: null,
    });
  }
}
```

```javascript
import { v4 as uuidv4 } from 'uuid';

export function correlationId(req, _res, next) {
  // Usar X-Correlation-Id si el cliente lo envía, o generar nuevo UUID
  const cid = req.headers['x-correlation-id'] || req.headers['x-request-id'] || uuidv4();
  req.correlationId = cid;
  req.id = cid; // Reutilizar como request ID también

  // Propagar a respuesta
  _res.setHeader('X-Correlation-Id', cid);
  next();
}
```

1. Modificar `src/middlewares/errorHandler.js` para incluir `correlation_id`:

```javascript
export function errorHandler(err, req, res, _next) {
  const statusCode = err.statusCode || 500;
  const code = err.code || 'INTERNAL_ERROR';
  const message = statusCode === 500 ? 'Error interno del servidor' : err.message;

  if (statusCode === 500) {
    req.log?.error({ err, reqId: req.id, correlationId: req.correlationId }, message);
  }

  return res.status(statusCode).json({
    success: false,
    error: { code, message, details: err.details || [] },
    correlation_id: req.correlationId || null,
  });
}
```
